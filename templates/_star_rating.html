{# Reusable star display widget. Pass `rating` (float 0-5) and `count` (int) #}
{% set full = rating | int %} {% set half = (rating - full) >= 0.4 %} {% for i in range(1, 6) %} {% if i <= full %} ★ {% elif half and i == full + 1 %} ★ {% else %} ★ {% endif %} {% endfor %} {% if count is defined %} ({{ count }}) {% endif %}
